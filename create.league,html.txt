<script>
// --- Leagues storage ---
let leagues = [];

// Create league
function createLeague() {
    const name = document.getElementById('leagueName').value.trim();
    const type = document.getElementById('leagueType').value.trim();
    const maxPlayers = document.getElementById('maxPlayers').value;
    const owner = document.getElementById('leagueOwner').value.trim();

    if(!name || !type || !maxPlayers || !owner){
        alert("‚ö†Ô∏è Fill all fields!");
        return;
    }

    leagues.push({name, type, maxPlayers, owner});
    showLeagues();

    // Clear input fields
    document.getElementById('leagueName').value = '';
    document.getElementById('leagueType').value = '';
    document.getElementById('maxPlayers').value = '';
    document.getElementById('leagueOwner').value = '';
}

// Show leagues
function showLeagues(){
    const div = document.getElementById('leagueList');
    if(leagues.length === 0){
        div.innerHTML = "<p>No leagues created yet.</p>";
        return;
    }

    let html = '';
    leagues.forEach(l => {
        html += `<div class="league-card">
                    <h3>üèÜ ${l.name}</h3>
                    <p>Type: ${l.type}</p>
                    <p>Max Players: ${l.maxPlayers}</p>
                    <p>Owner: ${l.owner}</p>
                 </div>`;
    });
    div.innerHTML = html;
}
